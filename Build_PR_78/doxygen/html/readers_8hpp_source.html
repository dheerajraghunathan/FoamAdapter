32template<typename FoamType>
33auto readVolBoundaryConditions(const NeoN::UnstructuredMesh& nfMesh, const FoamType& ofVolField)
34{
35 using type_container_t = typename type_map<FoamType>::container_type;
36 using type_primitive_t = typename type_map<FoamType>::mapped_type;
37
38 // get boundary as dictionary
39 OStringStream os;
40 ofVolField.boundaryField().writeEntries(os);
41 IStringStream is(os.str());
42 dictionary bDict(is);
43
44 std::map<std::string, std::function<void(NeoN::Dictionary&)>> patchInserter {
45 {"fixedGradient",
46 [](auto& dict)
47 {
48 dict.insert("type", std::string("fixedGradient"));
49 NeoN::TokenList tokenList = dict.template get<NeoN::TokenList>("value");
50 type_primitive_t fixedGradient = tokenList.get<type_primitive_t>(1);
51 dict.insert("fixedGradient", fixedGradient);
52 }},
53 {"zeroGradient",
54 [&](auto& dict)
55 {
56 dict.insert("type", std::string("fixedGradient"));
57 dict.insert("fixedGradient", type_primitive_t {});
58 }},
59 {"fixedValue",
60 [](auto& dict)
61 {
62 dict.insert("type", std::string("fixedValue"));
63 NeoN::TokenList tokenList = dict.template get<NeoN::TokenList>("value");
64 type_primitive_t fixedValue {};
65 if (std::is_same<type_primitive_t, NeoN::Vec3>::value)
66 {
67 NeoN::Vec3 tmpFixedValue {};
68 tmpFixedValue[0] = tokenList.get<int>(1);
69 tmpFixedValue[1] = tokenList.get<int>(2);
70 tmpFixedValue[2] = tokenList.get<int>(3);
71 dict.insert("fixedValue", tmpFixedValue);
72 }
73 else
74 {
75 fixedValue = tokenList.get<type_primitive_t>(1);
76 dict.insert("fixedValue", fixedValue);
77 }
78 }},
79 {"noSlip", // TODO specialize for vector
80 [](auto& dict)
81 {
82 dict.insert("type", std::string("fixedValue"));
83 dict.insert("fixedValue", type_primitive_t {});
84 }},
85 {"calculated", [](auto& dict) { dict.insert("type", std::string("calculated")); }},
86 {"extrapolatedCalculated",
87 [](auto& dict) { dict.insert("type", std::string("calculated")); }},
88 {"empty", [](auto& dict) { dict.insert("type", std::string("empty")); }}
89 };
90
91 int patchi = 0;
92 std::vector<fvcc::VolumeBoundary<type_primitive_t>> bcs;
93 for (const auto& bName : bDict.toc())
94 {
95 dictionary patchDict = bDict.subDict(bName);
96 NeoN::Dictionary neoPatchDict = Foam::readFoamDictionary(patchDict);
97 patchInserter[patchDict.get<word>("type")](neoPatchDict);
98 bcs.emplace_back(nfMesh, neoPatchDict, patchi);
99 patchi++;
100 }
101 return bcs;
102}

122template<typename FoamType>
123auto readSurfaceBoundaryConditions(
124 const NeoN::UnstructuredMesh& uMesh,
125 const FoamType& surfaceField
126)
127{
128 using type_container_t = typename type_map<FoamType>::container_type;
129 using type_primitive_t = typename type_map<FoamType>::mapped_type;
130
131 std::vector<fvcc::SurfaceBoundary<type_primitive_t>> bcs;
132
133 // get boundary as dictionary
134 OStringStream os;
135 surfaceField.boundaryField().writeEntries(os);
136 IStringStream is(os.str());
137 dictionary bDict(is);
138 int patchi = 0;
139
140 std::map<std::string, std::function<void(NeoN::Dictionary&)>> patchInserter {
141 {"fixedGradient", [](auto& dict) { dict.insert("type", std::string("fixedGradient")); }},
142 {"zeroGradient",
143 [&](auto& dict)
144 {
145 dict.insert("type", std::string("fixedGradient"));
146 dict.insert("fixedGradient", type_primitive_t {});
147 }},
148 {"fixedValue",
149 [](auto& dict)
150 {
151 dict.insert("type", std::string("fixedValue"));
152 dict.insert("fixedValue", type_primitive_t {});
153 }},
154 {"calculated", [](auto& dict) { dict.insert("type", std::string("calculated")); }},
155 {"empty", [](auto& dict) { dict.insert("type", std::string("empty")); }}
156 };
157
158 for (const auto& bName : bDict.toc())
159 {
160 dictionary patchDict = bDict.subDict(bName);
161 NeoN::Dictionary neoPatchDict;
162 patchInserter[patchDict.get<word>("type")](neoPatchDict);
163 bcs.push_back(fvcc::SurfaceBoundary<type_primitive_t>(uMesh, neoPatchDict, patchi));
164 patchi++;
165 }
166 return bcs;
167}

169template<typename FoamType>
170auto constructSurfaceField(
171 const NeoN::Executor exec,
172 const NeoN::UnstructuredMesh& nfMesh,
173 const FoamType& in
174)
175{
176 using type_container_t = typename type_map<FoamType>::container_type;
177 using type_primitive_t = typename type_map<FoamType>::mapped_type;
178 using foam_primitive_t = typename FoamType::cmptType;
179
180 type_container_t
181 out(exec, in.name(), nfMesh, std::move(readSurfaceBoundaryConditions(nfMesh, in)));
182
183 Field<foam_primitive_t> flattenedField(out.internalVector().size());
184 size_t nInternal = nfMesh.nInternalFaces();
185
186 forAll(in, facei)
187 {
188 flattenedField[facei] = convert(in[facei]);
189 }
190
191 label idx = nInternal;
192 forAll(in.boundaryField(), patchi)
193 {
194 const fvsPatchField<foam_primitive_t>& pin = in.boundaryField()[patchi];
195
196 forAll(pin, facei)
197 {
198 flattenedField[idx] = pin[facei];
199 idx++;
200 }
201 }
202 assert(idx == flattenedField.size());
203
204 out.internalVector() = fromFoamField(exec, flattenedField);
205 out.correctBoundaryConditions();
206
207 return out;
208}
